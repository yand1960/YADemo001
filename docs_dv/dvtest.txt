User Story:

Как фанат музыки, я хотел бы видеть
список доступных исполнителей и видеть даты их концертов,
чтобы иметь возможность забронировать билет.

Tables:

group: {id, name}
event: {id, date, group_id}
client: {id, name}
appointments: {id, client_id, event_id}

////////////////////////////////////////////

create table if not exists group -- список исполнителей
 (  id autoincrement not null primary key,
    name varchar(40) not null default ''
);

create table if not exists event -- список концертов
 (  id autoincrement not null primary key,
    date datetime not null,
    group_id varchar(40) not null default ''
);

create table if not exists client -- список зрителей
 (  id autoincrement not null primary key,
    date datetime not null,
    name varchar(40) not null default ''
);

create table if not exists appointments -- список заказов
 (  id autoincrement not null primary key,
    client_id integer not null,
    event_id varchar(40) not null
);
